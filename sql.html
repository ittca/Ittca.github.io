1- Criação, estruturação, introdução de dados e leitura de bd e suas tabelas

show databases; # mostra todas as bases de dados (bd)
create database nome_da_bd ; # criar um bd
drop database nome_da_bd ; # apaga uma bd
use nome_da_bd ; # seleciona a tabela que se pretende usar
show tables; # mostra as tabelas da bd
select * from nome_da_tabela ; # para ver todos os dados de certa tabela
drop table nome_da_tabela ; # apaga uma tabela


create table nome_da_tabela ( # cria uma tabela com 3 colunas
titulo_da_coluna tipo_de_dados ,
titulo_da_coluna tipo_de_dados
